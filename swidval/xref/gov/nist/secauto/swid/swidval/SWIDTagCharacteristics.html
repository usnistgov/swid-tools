27  package gov.nist.secauto.swid.swidval;
28  
29  import gov.nist.secauto.decima.xml.document.XMLDocument;
30  import gov.nist.secauto.decima.xml.document.XPathEvaluator;
31  import gov.nist.secauto.decima.xml.document.XPathNamespaceContext;
32  
33  import java.util.Objects;
34  
35  import javax.xml.xpath.XPathConstants;
36  import javax.xml.xpath.XPathExpressionException;
37  import javax.xml.xpath.XPathFactoryConfigurationException;
38  
39  public class SWIDTagCharacteristics {
40    /**
41     * Detects the SWID tag characteristics for the provided file.
42     * 
43     * @param doc
44     *          the file to generate characteristics for
45     * @return the detected SWID tag characteristics
46     * @throws XPathExpressionException
47     *           if an error occurred while evaluating the XPath expressions used to determine the
48     *           characteristics
49     * @throws XPathFactoryConfigurationException
50     *           if an error occurred while establishing the XPath evaluation environment
51     */
52    public static SWIDTagCharacteristics getSWIDTagCharacteristics(XMLDocument doc)
53        throws XPathExpressionException, XPathFactoryConfigurationException {
54      XPathEvaluator eval = doc.newXPathEvaluator();
55      XPathNamespaceContext nsContext = new XPathNamespaceContext();
56      nsContext.addNamespace("swid", "http://standards.iso.org/iso/19770/-2/2015/schema.xsd");
57      eval.setNamespaceContext(nsContext);
58  
59      boolean authoritative = eval.test("//swid:Entity[contains(@role,'tagCreator') and "
60          + "(contains(@role,'aggregator') or contains(@role,'distributor') or contains(@role,'licensor') or "
61          + "contains(@role,'softwareCreator'))]");
62      String tagTypeValue = eval.evaluateSingle(
63          "if (/swid:SoftwareIdentity[@patch='true']) then 'patch' "
64              + "else (if (/swid:SoftwareIdentity[@supplemental='true']) then 'supplemental' "
65              + "else (if (/swid:SoftwareIdentity[@corpus='true']) then 'corpus' else 'primary'))",
66          XPathConstants.STRING, null);
67      TagType tagType = TagType.lookup(tagTypeValue);
68      return new SWIDTagCharacteristics(tagType, authoritative);
69    }
70  
71    private final TagType tagType;
72    private final boolean authoritative;
73  
74    /**
75     * Construct a new SWID tag characteristics using the provided characteristics.
76     * 
77     * @param tagType
78     *          the type of the tag
79     * @param authoritative
80     *          <code>true</code> if the tag is authoritative, or <code>false</code> otherwise
81     */
82    public SWIDTagCharacteristics(TagType tagType, boolean authoritative) {
83      Objects.requireNonNull(tagType, "tagType");
84      this.tagType = tagType;
85      this.authoritative = authoritative;
86    }
87  
88    /**
89     * Retrieve the tag type.
90     * 
91     * @return the tagType
92     */
93    public TagType getTagType() {
94      return tagType;
95    }
96  
97    /**
98     * Determine if the tag is authoritative or not.
99     * 
100    * @return <code>true</code> if the tag is authoritative, or <code>false</code> otherwise
101    */
102   public boolean isAuthoritative() {
103     return authoritative;
104   }
105 }
