27  package gov.nist.secauto.swid.builder.resource;
28  
29  import java.io.IOException;
30  import java.security.NoSuchAlgorithmException;
31  import java.util.Collections;
32  import java.util.EnumMap;
33  import java.util.List;
34  import java.util.Map;
35  
36  public abstract class AbstractObjectBasedResourceEntry<ENTRY_TYPE> implements ResourceEntry {
37    private final Map<HashAlgorithm, List<Byte>> digests;
38    private final ENTRY_TYPE resource;
39  
40    public AbstractObjectBasedResourceEntry(ENTRY_TYPE resource) throws NoSuchAlgorithmException, IOException {
41      this.resource = resource;
42      this.digests = processDigests(resource);
43    }
44  
45    protected abstract EnumMap<HashAlgorithm, List<Byte>> processDigests(ENTRY_TYPE resource)
46        throws NoSuchAlgorithmException, IOException;
47  
48    protected ENTRY_TYPE getResource() {
49      return resource;
50    }
51  
52    @Override
53    public Map<HashAlgorithm, List<Byte>> getDigestValues() {
54      return Collections.unmodifiableMap(digests);
55    }
56  
57    @Override
58    public List<Byte> getDigestValue(HashAlgorithm key) throws NoSuchAlgorithmException {
59      List<Byte> retval = digests.get(key);
60      if (retval == null) {
61        throw new NoSuchAlgorithmException(key.getName());
62      }
63      return retval;
64    }
65  }
