27  package gov.nist.secauto.swid.builder;
28  
29  public enum KnownRole implements Role {
30    TAG_CREATOR(1, "tagCreator"),
31    SOFTWARE_CREATOR(2, "softwareCreator"),
32    AGGREGATOR(3, "aggregator"),
33    DISTRIBUTOR(4, "distributor"),
34    LICENSOR(5, "licensor");
35  
36    private final int index;
37    private final String name;
38  
39    private KnownRole(int index, String name) {
40      this.index = index;
41      this.name = name;
42      init(index, name);
43    }
44  
45    /**
46     * Get the integer index value for the role.
47     * 
48     * @return the index
49     */
50    @Override
51    public Integer getIndex() {
52      return index;
53    }
54  
55    /**
56     * Get the human-readable text value for the role.
57     * 
58     * @return the value
59     */
60    @Override
61    public String getName() {
62      return name;
63    }
64  }
