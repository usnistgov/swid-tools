27  package gov.nist.secauto.swid.builder.output;
28  
29  import com.fasterxml.jackson.core.JsonFactory;
30  import com.fasterxml.jackson.core.JsonGenerator;
31  import com.fasterxml.jackson.dataformat.cbor.CBORFactory;
32  
33  import gov.nist.secauto.swid.builder.Role;
34  import gov.nist.secauto.swid.builder.VersionScheme;
35  
36  import java.io.IOException;
37  
38  public class CBOROutputHandler extends AbstractJsonOutputHandler {
39  
40    public CBOROutputHandler() {
41      this(new CBORFactory());
42    }
43  
44    public CBOROutputHandler(JsonFactory jsonFactory) {
45      super(jsonFactory);
46    }
47  
48    @Override
49    protected void writeRole(JsonGenerator generator, Role role) throws IOException {
50      Integer index = role.getIndex();
51      if (index != null) {
52        generator.writeNumber(index);
53      } else {
54        generator.writeString(role.getName());
55      }
56    }
57  
58    @Override
59    protected void writeVersionScheme(JsonGenerator generator, VersionScheme versionScheme) throws IOException {
60      Integer index = versionScheme.getIndex();
61      if (index != null) {
62        generator.writeNumber(index);
63      } else {
64        generator.writeString(versionScheme.getName());
65      }
66    }
67  
68  }
