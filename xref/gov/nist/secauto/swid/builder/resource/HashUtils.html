48    /**
49     * Converts an array of bytes into a list of bytes.
50     * 
51     * @param bytes
52     *          the array of bytes to convert
53     * @return a list of bytes
54     */
55    public static List<Byte> toList(byte[] bytes) {
56      List<Byte> retval = new ArrayList<>(bytes.length);
57      for (byte b : bytes) {
58        retval.add(b);
59      }
60      return retval;
61    }

63    /**
64     * Converts a list of bytes into an array of bytes.
65     * 
66     * @param bytes
67     *          a list of bytes
68     * @return the array of bytes to convert
69     */
70    public static byte[] toArray(List<Byte> bytes) {
71      byte[] retval = new byte[bytes.size()];
72      for (int pos = 0; pos < bytes.size(); pos++) {
73        retval[pos] = bytes.get(pos);
74      }
75      return retval;
76    }

78    private static final char[] hexArray = "0123456789abcdef".toCharArray();

80    /**
81     * Converts an array of bytes into a hexadecimal string.
82     * 
83     * @param bytes
84     *          the array of bytes to convert
85     * @return a hexadecimal string
86     */
87    public static String toHexString(byte[] bytes) {
88      // Based on example from:
89      // http://stackoverflow.com/questions/9655181/how-to-convert-a-byte-array-to-a-hex-string-in-java
90      char[] hexChars = new char[bytes.length * 2];
91      for (int i = 0; i < bytes.length; i++) {
92        int value = bytes[i] & 0xFF;
93        hexChars[i * 2] = hexArray[value >>> 4];
94        hexChars[i * 2 + 1] = hexArray[value & 0x0F];
95      }
96      return new String(hexChars);
97    }

99    /**
100    * Converts an list of bytes into a hexadecimal string.
101    * 
102    * @param bytes
103    *          the list of bytes to convert
104    * @return a hexadecimal string
105    */
106   public static String toHexString(List<Byte> bytes) {
107     // Based on example from:
108     // http://stackoverflow.com/questions/9655181/how-to-convert-a-byte-array-to-a-hex-string-in-java
109     char[] hexChars = new char[bytes.size() * 2];
110     int pos = 0;
111     for (byte b : bytes) {
112       int value = b & 0xFF;
113       hexChars[pos * 2] = hexArray[value >>> 4];
114       hexChars[pos * 2 + 1] = hexArray[value & 0x0F];
115       ++pos;
116     }
117     return new String(hexChars);
118   }

120   public static byte[] toBytes(String hashHexBytes) {
121     try {
122       return Hex.decodeHex(hashHexBytes);
123     } catch (DecoderException e) {
124       throw new RuntimeException(e);
125     }

128   /**
129    * Generates a hash value, in the form of an array of bytes, by digesting a provided input stream
130    * based on the provided hash algorithm.
131    * 
132    * @param algorithm
133    *          the hash function to use
134    * @param file
135    *          the file to read bytes from
136    * @return an array of bytes representing a hash value
137    * @throws NoSuchAlgorithmException
138    *           if the selected hash function is not supported
139    * @throws IOException
140    *           if an error occured while reading the input stream
141    */
142   public static byte[] hash(HashAlgorithm algorithm, File file) throws NoSuchAlgorithmException, IOException {
143     return hash(algorithm, new BufferedInputStream(new FileInputStream(file)));
144   }

146   /**
147    * Generates a hash value, in the form of an array of bytes, by digesting a provided input stream
148    * based on the provided hash algorithm.
149    * 
150    * @param algorithm
151    *          the hash function to use
152    * @param is
153    *          the input stream to read bytes from
154    * @return an array of bytes representing a hash value
155    * @throws NoSuchAlgorithmException
156    *           if the selected hash function is not supported
157    * @throws IOException
158    *           if an error occured while reading the input stream
159    */
160   public static byte[] hash(HashAlgorithm algorithm, InputStream is) throws NoSuchAlgorithmException, IOException {
161     MessageDigest digest = MessageDigest.getInstance(algorithm.getName());
162     byte[] dataBytes = new byte[1024];
163     int nread = 0;
164 
165     while ((nread = is.read(dataBytes)) != -1) {
166       digest.update(dataBytes, 0, nread);
167     }
168 
169     return processDigest(algorithm, digest.digest());
170   }

172   /**
173    * Generates a hash value, in the form of an array of bytes, by digesting a provided bytes based on
174    * the provided hash algorithm.
175    * 
176    * @param algorithm
177    *          the hash function to use
178    * @param bytes
179    *          the bytes to digest
180    * @return an array of bytes representing a hash value
181    * @throws NoSuchAlgorithmException
182    *           if the selected hash function is not supported
183    */
184   public static byte[] hash(HashAlgorithm algorithm, byte[] bytes) throws NoSuchAlgorithmException {
185     MessageDigest digest = MessageDigest.getInstance(algorithm.getName());
186     digest.update(bytes);
187 
188     return processDigest(algorithm, digest.digest());
189   }

200    *         algorithm

202   public static byte[] processDigest(HashAlgorithm algorithm, byte[] mdbytes) {
203     int valueLength = algorithm.getValueLength() / 8;
204     if (valueLength < mdbytes.length) {
205       mdbytes = Arrays.copyOfRange(mdbytes, 0, valueLength);
206     }
